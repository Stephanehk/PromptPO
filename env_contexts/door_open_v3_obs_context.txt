"""
Structured observation for Meta-World MT1 task door-open-v3 (SawyerDoorEnvV3).

Meta-World builds observations in SawyerXYZEnv._get_obs (see metaworld/sawyer_xyz_env.py).
Door-specific state and success checks are in metaworld/envs/sawyer_door_v3.py.

Assumptions:
- env is from gym.make("Meta-World/MT1", env_name="door-open-v3", ...) after
  `import metaworld`.
- obs_np is the float vector returned by reset/step for this timestep.

Lives under direct_policy_learning/observations/.
"""

import numpy as np


class MetaWorldDoorOpenV3Observation:
    """
    door-open-v3 observation bundle for policies and reward code.

    Action space: Box(-1, 1, (4,), float32) — delta xyz + gripper
    (see env_contexts/door_open_v3_policy_context.txt).

    ---------------------------------------------------------------------------
    obs_vector — length 39 (default Sawyer MT1). Layout from SawyerXYZEnv._get_obs:
      - current frame (18): hand xyz, gripper, object/handle feature block (14)
      - previous frame (18)
      - goal (3): target related to the pull motion / handle interaction

    Door angle and handle geometry are reflected through the simulator state encoded in
    the observation and used inside evaluate_state (reward uses internal MuJoCo data as
    well as obs).

    Ground-truth reward delegates to SawyerDoorEnvV3.evaluate_state(obs, action).

    ---------------------------------------------------------------------------
    obs_vector — policy input.
    _env — internal only (evaluate_state for GT reward).
    """

    def __init__(self):
        self.obs_vector = None
        self._env = None

    def fill_from_env(self, env, obs_np):
        self.obs_vector = np.asarray(obs_np, dtype=np.float64).copy()
        self._env = env
